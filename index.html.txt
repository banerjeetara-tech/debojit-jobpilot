<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1">
  <title>JobPilot - Recovery Manager Roles</title>
  <script src="https://cdn.tailwindcss.com"></script>
  <link rel="preconnect" href="https://fonts.googleapis.com">
  <link href="https://fonts.googleapis.com/css2?family=Fraunces:opsz,wght@9..144,500;9..144,600&family=Hanken+Grotesk:wght@400;500;600;700&display=swap" rel="stylesheet">
  <style>
    body { font-family: 'Hanken Grotesk', sans-serif; }
    .font-display { font-family: 'Fraunces', serif; }
    * { -webkit-tap-highlight-color: transparent; }
    input, textarea { font-family: inherit; }
  </style>
</head>
<body class="bg-yellow-50" style="background: #F6F2EA;">
  <div id="app" class="min-h-screen flex flex-col"></div>

  <script>
    // ==================== CONFIG ====================
    const JSONBIN_ID = "6a2a8a2cda38895dfeac7eca";
    const JSONBIN_KEY = "$2a$10$SjTJd3D0tBhLZ5C8s.0Ef.5vCebu0ctZs9Fwhrxkss9DK2YsYbwru";
    
    // Colors
    const C = {
      paper: "#F6F2EA",
      card: "#FFFFFF",
      ink: "#1C1B17",
      sub: "#6F6B61",
      line: "#E9E2D5",
      accent: "#0E5A4E",
      accentSoft: "#E3EFEB",
      amber: "#9A5B12",
      amberSoft: "#F6EBD9",
      green: "#1F7A4D",
      greenSoft: "#E2F1E8",
    };

    // ==================== STATE ====================
    let jobs = [];
    let view = "jobs"; // jobs, add, tracker
    let openJobId = null;
    let lastRefresh = new Date();

    // ==================== LOAD FROM JSONBIN ====================
    async function loadJobs() {
      try {
        const res = await fetch(
          `https://api.jsonbin.io/v3/b/${JSONBIN_ID}/latest?meta=false`,
          { headers: { "X-Master-Key": JSONBIN_KEY } }
        );
        if (!res.ok) {
          console.error("JSONBin error:", res.status);
          return [];
        }
        const data = await res.json();
        return Array.isArray(data) ? data : (data.jobs || []);
      } catch (e) {
        console.error("Load error:", e.message);
        return [];
      }
    }

    // ==================== SAVE TO JSONBIN ====================
    async function saveJobs() {
      try {
        const res = await fetch(
          `https://api.jsonbin.io/v3/b/${JSONBIN_ID}`,
          {
            method: "PUT",
            headers: {
              "X-Master-Key": JSONBIN_KEY,
              "Content-Type": "application/json",
            },
            body: JSON.stringify({ jobs, lastUpdated: new Date().toISOString() }),
          }
        );
        return res.ok;
      } catch (e) {
        console.error("Save error:", e.message);
        return false;
      }
    }

    // ==================== LOAD ON START ====================
    async function init() {
      const loaded = await loadJobs();
      jobs = loaded.length > 0 ? loaded : [];
      render();
      
      // Auto-refresh every 5 minutes
      setInterval(async () => {
        const latest = await loadJobs();
        if (latest.length > 0) {
          jobs = latest;
          lastRefresh = new Date();
          render();
        }
      }, 5 * 60 * 1000);
    }

    // ==================== RENDER ====================
    function render() {
      const html = `
        <header style="background: ${C.card}; border-bottom: 1px solid ${C.line};" class="sticky top-0 z-50 py-4 px-4 flex items-center justify-between">
          <div class="flex items-center gap-2">
            <div class="font-display font-semibold">JobPilot</div>
            <div class="text-[11px]" style="color: ${C.sub}">Debojit - Recovery Manager</div>
          </div>
          <button onclick="refreshJobs()" style="background: ${C.accentSoft}" class="px-3 py-2 rounded-lg text-sm font-semibold" style="color: ${C.accent}">🔄</button>
        </header>

        <nav style="background: ${C.card}; border-bottom: 1px solid ${C.line};" class="sticky top-16 z-40 px-4 flex gap-4">
          <button onclick="setView('jobs')" style="border-bottom: ${view === 'jobs' ? `2px solid ${C.accent}` : 'none'}; color: ${view === 'jobs' ? C.accent : C.sub};" class="py-3 text-sm font-semibold">Jobs (${jobs.length})</button>
          <button onclick="setView('add')" style="border-bottom: ${view === 'add' ? `2px solid ${C.accent}` : 'none'}; color: ${view === 'add' ? C.accent : C.sub};" class="py-3 text-sm font-semibold">Add</button>
          <button onclick="setView('tracker')" style="border-bottom: ${view === 'tracker' ? `2px solid ${C.accent}` : 'none'}; color: ${view === 'tracker' ? C.accent : C.sub};" class="py-3 text-sm font-semibold">Tracker</button>
        </nav>

        <main class="flex-1 p-4 max-w-2xl mx-auto w-full pb-20">
          ${renderView()}
        </main>

        <footer style="background: ${C.card}; border-top: 1px solid ${C.line};" class="fixed bottom-0 left-0 right-0 text-center text-[11px] py-2" style="color: ${C.sub}">
          Last refreshed: ${lastRefresh.toLocaleTimeString()} | Auto-refresh: ✓ ON (5 min)
        </footer>
      `;
      document.getElementById("app").innerHTML = html;
    }

    function renderView() {
      if (view === "jobs") return renderJobsList();
      if (view === "add") return renderAddJob();
      if (view === "tracker") return renderTracker();
    }

    function renderJobsList() {
      if (jobs.length === 0) {
        return `<div style="background: ${C.card}; border: 1px solid ${C.line};" class="rounded-2xl p-8 text-center">No jobs yet. Check n8n or add manually.</div>`;
      }
      
      return jobs.map((job, i) => `
        <div style="background: ${C.card}; border: 1px solid ${C.line};" class="rounded-2xl mb-3 overflow-hidden">
          <button onclick="toggleJob(${i})" class="w-full p-4 text-left hover:opacity-75 flex justify-between items-start">
            <div class="flex-1">
              <div style="color: ${C.ink}" class="font-semibold text-sm">${job.title}</div>
              <div style="color: ${C.sub}" class="text-xs mt-1">${job.company} • ${job.location}</div>
              ${job.score ? `
                <div class="mt-2 flex gap-2 items-center">
                  <span style="background: ${job.score === 'HIGH' ? C.greenSoft : job.score === 'MEDIUM' ? C.amberSoft : C.line}; color: ${job.score === 'HIGH' ? C.green : job.score === 'MEDIUM' ? C.amber : C.sub};" class="text-xs font-bold px-2 py-0.5 rounded-full">${job.score}</span>
                  <span style="color: ${C.sub}" class="text-xs">${job.reason}</span>
                </div>
              ` : ''}
            </div>
            <div>${openJobId === i ? '▲' : '▼'}</div>
          </button>
          
          ${openJobId === i ? `
            <div style="border-top: 1px solid ${C.line};" class="px-4 py-4 space-y-3">
              ${job.jd ? `
                <div>
                  <div style="color: ${C.sub}" class="text-xs font-bold mb-2">JOB DESCRIPTION</div>
                  <div style="color: ${C.ink}" class="text-[12px] leading-relaxed">${job.jd.slice(0, 400)}...</div>
                </div>
              ` : ''}
              
              <div class="flex gap-2 flex-wrap pt-2">
                <button onclick="markApplied(${i})" style="background: ${job.status === 'applied' ? C.greenSoft : C.accentSoft}; color: ${job.status === 'applied' ? C.green : C.accent};" class="flex-1 py-2 rounded-lg text-xs font-semibold">
                  ${job.status === 'applied' ? '✓ Applied' : 'Mark Applied'}
                </button>
                <button onclick="deleteJob(${i})" style="background: #F5E3E3; color: #A93F3F;" class="px-3 py-2 rounded-lg text-xs font-semibold">Delete</button>
                ${job.url ? `<a href="${job.url}" target="_blank" style="background: ${C.card}; border: 1px solid ${C.line}; color: ${C.accent};" class="px-3 py-2 rounded-lg text-xs font-semibold">→</a>` : ''}
              </div>
            </div>
          ` : ''}
        </div>
      `).join('');
    }

    function renderAddJob() {
      return `
        <div class="space-y-4">
          <div class="font-display text-xl">Add Job</div>
          <div style="color: ${C.sub}" class="text-sm">Manually add recovery manager jobs</div>
          
          <div>
            <label style="color: ${C.sub}" class="text-xs font-semibold block mb-1">Job Title *</label>
            <input id="title" type="text" placeholder="Recovery Manager" style="background: ${C.card}; border: 1px solid ${C.line}; color: ${C.ink};" class="w-full px-3 py-2 rounded-lg text-sm">
          </div>
          
          <div>
            <label style="color: ${C.sub}" class="text-xs font-semibold block mb-1">Company *</label>
            <input id="company" type="text" placeholder="HDFC Bank" style="background: ${C.card}; border: 1px solid ${C.line}; color: ${C.ink};" class="w-full px-3 py-2 rounded-lg text-sm">
          </div>
          
          <div>
            <label style="color: ${C.sub}" class="text-xs font-semibold block mb-1">Location</label>
            <input id="location" type="text" placeholder="Kolkata, West Bengal" style="background: ${C.card}; border: 1px solid ${C.line}; color: ${C.ink};" class="w-full px-3 py-2 rounded-lg text-sm">
          </div>
          
          <div>
            <label style="color: ${C.sub}" class="text-xs font-semibold block mb-1">Job URL</label>
            <input id="url" type="text" placeholder="https://..." style="background: ${C.card}; border: 1px solid ${C.line}; color: ${C.ink};" class="w-full px-3 py-2 rounded-lg text-sm">
          </div>
          
          <div>
            <label style="color: ${C.sub}" class="text-xs font-semibold block mb-1">Job Description *</label>
            <textarea id="jd" rows="6" placeholder="Paste job description..." style="background: ${C.card}; border: 1px solid ${C.line}; color: ${C.ink};" class="w-full px-3 py-2 rounded-lg text-sm resize-none"></textarea>
          </div>
          
          <button onclick="addJob()" style="background: ${C.accent}; color: white;" class="w-full py-3 rounded-lg font-semibold text-sm">Save Job</button>
        </div>
      `;
    }

    function renderTracker() {
      const applied = jobs.filter(j => j.status === 'applied').length;
      const pending = jobs.length - applied;
      
      return `
        <div class="space-y-5">
          <div class="font-display text-xl">Tracker</div>
          
          <div class="grid grid-cols-3 gap-3">
            <div style="background: ${C.card}; border: 1px solid ${C.line};" class="rounded-xl p-4 text-center">
              <div style="color: ${C.ink};" class="font-display text-2xl">${jobs.length}</div>
              <div style="color: ${C.sub};" class="text-xs mt-1">Total</div>
            </div>
            <div style="background: ${C.greenSoft}; border: 1px solid ${C.line};" class="rounded-xl p-4 text-center">
              <div style="color: ${C.green};" class="font-display text-2xl">${applied}</div>
              <div style="color: ${C.sub};" class="text-xs mt-1">Applied</div>
            </div>
            <div style="background: ${C.amberSoft}; border: 1px solid ${C.line};" class="rounded-xl p-4 text-center">
              <div style="color: ${C.amber};" class="font-display text-2xl">${pending}</div>
              <div style="color: ${C.sub};" class="text-xs mt-1">Pending</div>
            </div>
          </div>
          
          ${jobs.filter(j => j.status === 'applied').length === 0 ? 
            `<div style="background: ${C.card}; border: 1px solid ${C.line};" class="rounded-xl p-4 text-center"><div style="color: ${C.sub};" class="text-sm">Mark jobs as Applied to track progress</div></div>` 
            : `
            <div>
              <div style="color: ${C.sub};" class="text-xs font-bold mb-3">APPLICATIONS</div>
              ${jobs.filter(j => j.status === 'applied').map(j => `
                <div style="background: ${C.card}; border: 1px solid ${C.line};" class="rounded-lg p-3 mb-2">
                  <div style="color: ${C.ink};" class="text-sm font-semibold">${j.title}</div>
                  <div style="color: ${C.sub};" class="text-xs">${j.company}</div>
                </div>
              `).join('')}
            </div>
          `}
        </div>
      `;
    }

    // ==================== FUNCTIONS ====================
    function setView(v) { view = v; render(); }
    function toggleJob(i) { openJobId = openJobId === i ? null : i; render(); }
    
    function addJob() {
      const title = document.getElementById('title').value.trim();
      const company = document.getElementById('company').value.trim();
      const location = document.getElementById('location').value.trim();
      const url = document.getElementById('url').value.trim();
      const jd = document.getElementById('jd').value.trim();
      
      if (!title || !company || !jd) {
        alert('Please fill in all required fields');
        return;
      }
      
      jobs.unshift({
        id: 'j' + Date.now(),
        title,
        company,
        location,
        url,
        jd,
        score: 'MEDIUM',
        reason: 'Manual entry',
        status: 'saved',
        dateAdded: new Date().toISOString().slice(0, 10),
      });
      
      document.getElementById('title').value = '';
      document.getElementById('company').value = '';
      document.getElementById('location').value = '';
      document.getElementById('url').value = '';
      document.getElementById('jd').value = '';
      
      saveJobs();
      view = 'jobs';
      openJobId = null;
      render();
    }
    
    function markApplied(i) {
      jobs[i].status = jobs[i].status === 'applied' ? 'saved' : 'applied';
      jobs[i].dateApplied = jobs[i].status === 'applied' ? new Date().toISOString().slice(0, 10) : null;
      saveJobs();
      render();
    }
    
    function deleteJob(i) {
      jobs.splice(i, 1);
      saveJobs();
      openJobId = null;
      render();
    }
    
    async function refreshJobs() {
      const latest = await loadJobs();
      if (latest.length > 0) {
        jobs = latest;
        lastRefresh = new Date();
        render();
      }
    }

    // Start
    init();
  </script>
</body>
</html>
